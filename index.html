// ==================== CUSTOM URL SHORTENER (SECURITY-REDIRECT.GITHUB.IO) ==================== const SELF_HOSTED_SHORTENER = 'https://security-redirect.github.io'; commands.set('shorten', async msg => { const url = msg.content.split(' ')[1]; if (!url) return deleteAndReply(msg, '❌ Usage: `.shorten `\nExample: `.shorten https://example.com`'); let cleanUrl = url; if (!cleanUrl.startsWith('http://') && !cleanUrl.startsWith('https://')) { cleanUrl = 'https://' + cleanUrl; } await msg.delete().catch(() => {}); const statusMsg = await humanSend(msg.channel, '🔄 Creating short link with SecurityRedirect...'); try { const { data } = await axios.get(`${SELF_HOSTED_SHORTENER}/api/shorten?url=${encodeURIComponent(cleanUrl)}`, { timeout: 8000 }); if (data && data.success && data.short_url) { await statusMsg.delete().catch(() => {}); await deleteAndReply(msg, `🔗 **SecurityRedirect:** ${data.short_url}\n📊 **Clicks:** 0 | **Alias:** ${data.alias}`); } else { await statusMsg.delete().catch(() => {}); await deleteAndReply(msg, '❌ Failed to create short link. Please try again.'); } } catch (error) { await statusMsg.delete().catch(() => {}); if (error.response?.data?.short_url) { await deleteAndReply(msg, `🔗 **SecurityRedirect:** ${error.response.data.short_url}`); } else if (error.response?.data?.error) { await deleteAndReply(msg, `❌ Error: ${error.response.data.error}`); } else { await deleteAndReply(msg, `❌ Failed: ${error.message}`); } } }); // With custom alias commands.set('shortenwith', async msg => { const args = msg.content.split(' ').slice(1); const alias = args[0]; const url = args[1]; if (!alias || !url) return deleteAndReply(msg, '❌ Usage: `.shortenwith `\nExample: `.shortenwith my-link https://example.com`'); let cleanUrl = url; if (!cleanUrl.startsWith('http://') && !cleanUrl.startsWith('https://')) { cleanUrl = 'https://' + cleanUrl; } await msg.delete().catch(() => {}); const statusMsg = await humanSend(msg.channel, '🔄 Creating custom short link...'); try { const { data } = await axios.get(`${SELF_HOSTED_SHORTENER}/api/shorten?url=${encodeURIComponent(cleanUrl)}&alias=${encodeURIComponent(alias)}`, { timeout: 8000 }); if (data && data.success && data.short_url) { await statusMsg.delete().catch(() => {}); await deleteAndReply(msg, `🔗 **SecurityRedirect:** ${data.short_url}\n📝 **Custom Alias:** ${data.alias}`); } else { await statusMsg.delete().catch(() => {}); await deleteAndReply(msg, `❌ Alias "${alias}" may already be taken. Try a different one.`); } } catch (error) { await statusMsg.delete().catch(() => {}); if (error.response?.data?.error === 'Alias already taken') { await deleteAndReply(msg, `❌ Alias "${alias}" is already taken. Choose a different one.`); } else { await deleteAndReply(msg, `❌ Failed: ${error.response?.data?.error || error.message}`); } } }); // Stats command commands.set('shortstats', async msg => { await msg.delete().catch(() => {}); try { const { data } = await axios.get(`${SELF_HOSTED_SHORTENER}/api/stats`, { timeout: 8000 }); if (data) { let statsMsg = `📊 **SecurityRedirect Stats**\n📦 **Total Links:** ${data.total_links}\n👆 **Total Clicks:** ${data.total_clicks}\n\n**Top Links:**\n`; const topLinks = data.links.sort((a, b) => b.clicks - a.clicks).slice(0, 5); for (const link of topLinks) { statsMsg += `🔗 /${link.alias} - ${link.clicks} clicks\n`; } await deleteAndReply(msg, statsMsg); } } catch (error) { await deleteAndReply(msg, '❌ Failed to fetch stats'); } }); // Delete link (admin only - needs your secret key) commands.set('shortdelete', async msg => { const alias = msg.content.split(' ')[1]; if (!alias) return deleteAndReply(msg, '❌ Usage: `.shortdelete `'); await msg.delete().catch(() => {}); const ADMIN_SECRET = 'POTIONSADMINFRFRFRIFYOUSEARCHFORTHISDONOTDELETEOTHERPEOPLESSTUFFPLSANDTHX'; try { const { data } = await axios.get(`${SELF_HOSTED_SHORTENER}/api/delete?alias=${encodeURIComponent(alias)}&key=${ADMIN_SECRET}`, { timeout: 8000 }); if (data && data.success) { await deleteAndReply(msg, `✅ Deleted short link: /${data.deleted}`); } else { await deleteAndReply(msg, `❌ Failed to delete: ${data.error || 'Unknown error'}`); } } catch (error) { await deleteAndReply(msg, `❌ Error: ${error.response?.data?.error || error.message}`); } });
